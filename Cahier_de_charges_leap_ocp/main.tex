% ========== 1. CONTEXTE ==========
\section{Contexte et pr\'esentation du projet}

\subsection{Contexte g\'en\'eral}
Dans le cadre de la gestion de projets au sein d'\textbf{OCP Solutions}, le suivi de l'avancement des projets repose souvent sur des processus manuels et fragment\'es. Les chefs de projet doivent consulter plusieurs sources d'information --- outils de gestion de t\^aches, d\'ep\^ots Git, comptes rendus de r\'eunions --- pour \'evaluer l'\'etat d'avancement r\'eel d'un projet.

Cette approche pr\'esente plusieurs limites : elle est chronophage, sujette aux erreurs et ne permet pas une d\'etection proactive des risques.

\subsection{Probl\'ematique}
Les principales difficult\'es identifi\'ees sont :
\begin{itemize}[leftmargin=2cm, itemsep=4pt]
    \item[\textcolor{accent}{$\blacktriangleright$}] Manque de visibilit\'e en temps r\'eel sur l'avancement des projets
    \item[\textcolor{accent}{$\blacktriangleright$}] D\'etection tardive des retards et des surcharges de travail
    \item[\textcolor{accent}{$\blacktriangleright$}] Informations dispers\'ees entre plusieurs outils et formats
    \item[\textcolor{accent}{$\blacktriangleright$}] Absence d'analyse automatique des comptes rendus de r\'eunions
    \item[\textcolor{accent}{$\blacktriangleright$}] Difficult\'e \`a identifier les risques de mani\`ere proactive
\end{itemize}

\subsection{Objectifs du projet}
Le projet vise \`a d\'evelopper une \textbf{plateforme intelligente de suivi de projets} capable d'analyser automatiquement l'avancement d'un projet \`a partir de trois sources de donn\'ees principales :

L'application fournira un \textbf{tableau de bord synth\'etique} avec des indicateurs de progression et des \textbf{alertes automatiques}.

\section{P\'erim\`etre du projet}

\subsection{P\'erim\`etre inclus}
\begin{itemize}[leftmargin=2cm, itemsep=4pt]
    \item[\textcolor{secondary}{\checkmark}] Suivi automatique de l'avancement des projets
    \item[\textcolor{secondary}{\checkmark}] Int\'egration avec les outils de gestion de t\^aches (Jira, Trello ou autre)

    \item[\textcolor{secondary}{\checkmark}] Analyse automatique des comptes rendus de r\'eunions par IA/NLP
    \item[\textcolor{secondary}{\checkmark}] Syst\`eme d'alertes pour les retards et surcharges

\end{itemize}

    \item[\textcolor{red!70!black}{$\times$}] Gestion de t\^aches (ne remplace pas Jira/Trello)
    \item[\textcolor{red!70!black}{$\times$}] H\'ebergement de code source (ne remplace pas Git)
    \item[\textcolor{red!70!black}{$\times$}] Gestion documentaire compl\`ete
    \item[\textcolor{red!70!black}{$\times$}] Application mobile (sauf indication contraire du client)
\end{itemize}

% ========== 4. BESOINS FONCTIONNELS ==========
\section{Besoins fonctionnels}

\subsection{F1 -- Suivi d'avancement automatique}
\begin{itemize}[leftmargin=2cm, itemsep=3pt]
    \item[\textcolor{accent}{$\blacktriangleright$}] Collecte automatique des donn\'ees depuis les outils de t\^aches via API
    \item[\textcolor{accent}{$\blacktriangleright$}] Analyse des commits Git (fr\'equence, volume, messages)
    \item[\textcolor{accent}{$\blacktriangleright$}] Extraction automatique d'informations cl\'es des comptes rendus
    \item[\textcolor{accent}{$\blacktriangleright$}] Calcul d'un score d'avancement global par projet
    \item[\textcolor{accent}{$\blacktriangleright$}] Historique de l'avancement dans le temps
\end{itemize}

\subsection{F2 -- Alertes retard / surcharge}
\begin{itemize}[leftmargin=2cm, itemsep=3pt]
    \item[\textcolor{accent}{$\blacktriangleright$}] D\'etection automatique des t\^aches en retard par rapport aux deadlines
    \item[\textcolor{accent}{$\blacktriangleright$}] D\'etection de la surcharge de travail par membre d'\'equipe
    \item[\textcolor{accent}{$\blacktriangleright$}] Envoi de notifications (email, in-app, ou autre canal \`a d\'efinir)
    \item[\textcolor{accent}{$\blacktriangleright$}] Configuration des seuils d'alerte par le chef de projet
    \item[\textcolor{accent}{$\blacktriangleright$}] Classification des alertes par niveau de criticit\'e (faible, moyen, critique)
\end{itemize}

\subsection{F3 -- Dashboard interactif}
\begin{itemize}[leftmargin=2cm, itemsep=3pt]
    \item[\textcolor{accent}{$\blacktriangleright$}] Vue globale de la progression du projet (pourcentage d'avancement)
    \item[\textcolor{accent}{$\blacktriangleright$}] Indicateurs de risques identifi\'es
    \item[\textcolor{accent}{$\blacktriangleright$}] R\'epartition de la charge par membre de l'\'equipe
    \item[\textcolor{accent}{$\blacktriangleright$}] Graphiques d'\'evolution temporelle
    \item[\textcolor{accent}{$\blacktriangleright$}] Filtres par projet, \'equipe, p\'eriode
    \item[\textcolor{accent}{$\blacktriangleright$}] Vue synth\'etique pour la direction
\end{itemize}

\subsection{F4 -- Analyse intelligente des comptes rendus (IA/NLP)}
\begin{itemize}[leftmargin=2cm, itemsep=3pt]
    \item[\textcolor{accent}{$\blacktriangleright$}] Extraction automatique des d\'ecisions prises
    \item[\textcolor{accent}{$\blacktriangleright$}] Identification des actions \`a r\'ealiser et leurs responsables
    \item[\textcolor{accent}{$\blacktriangleright$}] D\'etection des blocages et risques mentionn\'es
    \item[\textcolor{accent}{$\blacktriangleright$}] R\'esum\'e automatique des r\'eunions
    \item[\textcolor{accent}{$\blacktriangleright$}] Support multilingue (fran\c{c}ais et anglais \`a minima)
\end{itemize}

\subsection{F5 -- Int\'egration Git}
\begin{itemize}[leftmargin=2cm, itemsep=3pt]
    \item[\textcolor{accent}{$\blacktriangleright$}] Connexion aux d\'ep\^ots Git via API
    \item[\textcolor{accent}{$\blacktriangleright$}] Analyse de l'activit\'e des commits (fr\'equence, contributeurs)
    \item[\textcolor{accent}{$\blacktriangleright$}] Corr\'elation entre les commits et les t\^aches associ\'ees
    \item[\textcolor{accent}{$\blacktriangleright$}] D\'etection des p\'eriodes d'inactivit\'e
\end{itemize}

\subsection{F6 -- Gestion des utilisateurs}
\begin{itemize}[leftmargin=2cm, itemsep=3pt]
    \item[\textcolor{accent}{$\blacktriangleright$}] Authentification s\'ecuris\'ee
    \item[\textcolor{accent}{$\blacktriangleright$}] Gestion des r\^oles (administrateur, chef de projet, membre, lecteur)
    \item[\textcolor{accent}{$\blacktriangleright$}] Gestion des projets et affectation des membres
\end{itemize}
